#include <iostream>
using namespace std;
int main() {
    int a, b;
    cin >> a >> b;

    cout << a - b << endl;  // Subtraction
    cout << a * b << endl;  // Multiplication
    cout << a % b << endl;  // Modulo
    cout << a / b << endl;  // Division

    return 0;
}
